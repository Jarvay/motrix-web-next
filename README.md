## Usage
`docker-compose.yml`
```yaml
services:
  motrix-web-next:
    image: jarvay/motrix-web-next
    container_name: motrix-web-next
    restart: unless-stopped
    ports:
      - "22077:22077"
    volumes:
      - ./data:/data
      - ./downloads:/home/Downloads
    environment:
      - TZ=Asia/Shanghai
      # - PORT=22077              # default in Dockerfile
      # - HOST=0.0.0.0            # default in Dockerfile
      # - DATA_DIR=/data          # default in Dockerfile
      # - RPC_PORT=29100          # default in Dockerfile
```

```shell
docker-compose up -d
```
